dwm keybindings:

Super + p		 # run dmenu 
Super + Ctrl + t	 # run terminal 
Super + w		 # run browser 
Super + Ctrl + d	 # run discord 
Super + Ctrl + k	 # run keybind 
Super + b		 # toggle bar 
Super + j		 # focus on next window 
Super + k		 # focus on previous window 
Super + i		 # increase amount of masters 
Super + d		 # decrease amount of masters 
Super + h		 # increase size of slave windows 
Super + l		 # increase size of master window
Super + Return	 # replace master with selected window      
Super + Tab		 # go to last workspace 
Super + q		 # close window 
Super + t		 # tiling layout 
Super + f		 # floating layout 
Super + m		 # monocle layout 
Super + space		 # use last layout 
Super + Shift + space	 # use last layout only for selected window
Super + Ctrl + f	 # toggle fullscreen   
Super + 0		 # view all windows on one workspace                      
Super + Shift + 0	 # turn a window sticky                      
Super + ,		 # focus on previous monitor 
Super + .		 # focus on next monitor 
Super + Shift + ,	 # send window to previous screen          
Super + Shift + .	 # send window to next screen          
Super + →		 # next workspace 
Super + ←		 # previous workspace 
Super + F11		 # decrease volume 
Super + F9		 # mute sound 
Super + F12		 # increase volume 
Super + 1		 # go to workspace 1 
Super + 2		 # go to workspace 2 
Super + 3		 # go to workspace 3 
Super + 4		 # go to workspace 4 
Super + 5		 # go to workspace 5 
Super + 6		 # go to workspace 6 
Super + 7		 # go to workspace 7 
Super + 8		 # go to workspace 8 
Super + 9		 # go to workspace 9 
Super + Ctrl + q	 # quit dwm
